Endpoints:

- POST - /orders
- GET - /orders/{orderId}
- PUT - /orders/{orderId}


SQL:

CREATE DATABASE ordering;

CREATE TABLE order_headers
(
  id                UUID PRIMARY KEY,
  customer_id       UUID,
  last_updated      TIMESTAMP DEFAULT NOW()
);

CREATE TABLE orderLines
(
  id                UUID REFERENCES order_headers(id),
  item_number       INTEGER,
  quantity          INTEGER
);
